create table Autorid (
Autori_kood INT PRIMARY KEY AUTO_INCREMENT,
Nimi varchar(25) not null,
Perenimi varchar(30) not null,
Sunniaasta date,
Sugu varchar(30) not null);

create table Keel (
Keel_kood INT PRIMARY KEY AUTO_INCREMENT,
Keel varchar(30) not null);

create table Raamatud (
Raamatu_kood INT PRIMARY KEY AUTO_INCREMENT,
Nimetus varchar(25) not null,
V_aasta date,
Hind float,
Autor int,
Keel int,
foreign key (Autor) references Autorid(Autori_kood),
foreign key (Keel) references Keel(Keel_kood));

create table Ostukorv (
Ostu_id INT PRIMARY KEY AUTO_INCREMENT,
Raamat int,
Hind float,
Kogus int,
Linn int,
foreign key (Raamat) references raamatud(Raamatud_kood),
foreign key (Linn) references linn(linn_id));
