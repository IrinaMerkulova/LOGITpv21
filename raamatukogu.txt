CREATE TABLE Autorid (
Autori_kood int PRIMARY key AUTO_INCREMENT,
Nimi varchar(20) not null,
Perenimi varchar(20) not null,
Synniaasta date,
Sugu varchar(10) not null);

Create table Keel (
Keel_kood int primary key identity(1,1),
Keel varchar(10) not null);
select * from Keel;

create table Raamatud (
Raamatu_kood int primary key AUTO_INCREMENT,
Nimetus varchar(20) not null,
Autor int,
V_aasta date,
Keel int,
Hind decimal(5,2),
foreign key (Autor) references Autorid(Autori_kood),
foreign key (Keel) references Keel(Keel_kood)
);

create table Klient (
Kliendi_kood int primary key identity(1,1),
Nimi varchar(10) not null,
Perenimi varchar(10) not null,
Isikukood varchar(15) not null,
Aadress varchar(15) not null);
