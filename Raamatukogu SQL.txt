CREATE TABLE Autorid( 
Autori_kood int PRIMARY KEY AUTO_INCREMENT,
Nimi varchar(30),
Perenimi varchar(30),
Sunniaasta date,
Sugu varchar(10) 
);

CREATE TABLE Keel(
Keel_kood int PRIMARY KEY AUTO_INCREMENT,
Keel varchar(30) UNIQUE
);



CREATE TABLE Raamatud(
Raamatu_kood int PRIMARY KEY AUTO_INCREMENT,
Nimetus varchar(50),
Autor int,
V_aasta int,
Keel int,
Hind decimal(5,2),
foreign key (Autor) references Autorid(Autori_kood),
foreign key (Keel) references Keel(Keel_kood)
);



CREATE TABLE Klient(
Kliendi_kood int PRIMARY KEY AUTO_INCREMENT,
Nimi varchar(30),
Perenimi varchar(30),
Isikukood Bigint,
Aadress varchar(50)
);



CREATE TABLE Laenutus(
Laenutuse_kood int PRIMARY KEY AUTO_INCREMENT,
Klient int,
Raamat int,
Algus_kuupaev date,
Lopp_kuupaev date,
foreign key (Klient) references Klient(Kliendi_kood),
foreign key (Raamat) references Raamatud(Raamatu_kood)
);
